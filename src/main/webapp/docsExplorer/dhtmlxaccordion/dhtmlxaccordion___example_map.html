	<title>Attach Google Maps to Accordion</title>
	<link rel="stylesheet" type="text/css" href="codebase/dhtmlxaccordion.css">
	<link rel="stylesheet" type="text/css" href="codebase/skins/dhtmlxaccordion_dhx_skyblue.css">
	<script src="codebase/dhtmlxcommon.js"></script>
	<script src="codebase/dhtmlxaccordion.js"></script>
 
	<script src="codebase/dhtmlxcontainer.js"></script>
	<script type="text/javascript" src="http://maps.google.com/maps/api/js?sensor=false"></script>

<body onload="doOnLoad();">
	<div id="parentId" style="position: relative; top: 20px; left: 20px; width: 600px; height: 400px; aborder: #B5CDE4 1px solid;"></div>
 
<script>
	var dhxAccord, GMaps;
	function doOnLoad() {
		dhxAccord = new dhtmlXAccordion("parentId");
		dhxAccord.addItem("a", "www.google.com");
		dhxAccord.cells("a").setWidth(120);
		GMaps = dhxAccord.cells("a").attachMap();
 
	}
</script>
