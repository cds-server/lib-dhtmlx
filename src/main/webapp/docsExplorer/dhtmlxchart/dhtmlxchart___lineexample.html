	<link rel="STYLESHEET" type="text/css" href="../codebase/dhtmlxchart.css">
        <script>
	var data = [
		{ sales:"3.8", year:"2001" },
		{ sales:"3.4", year:"2002" },
		{ sales:"4.1", year:"2003" },
		{ sales:"4.3", year:"2004" },
		{ sales:"9.9", year:"2005" },
		{ sales:"7.4", year:"2006" },
		{ sales:"9.0", year:"2007" },
		{ sales:"7.3", year:"2008" },
		{ sales:"4.8", year:"2009" }
	];
 
        window.onload = function(){
	   var lineChart =  new dhtmlXChart({
		view:"line",
		container:"chart_container",
	        value:"#sales#",
		label:"#year#",
                tooltip: "#year#",
                item:{
                    borderColor:"#ffffff",
                    color:"#000000"
                },
                line:{
                     color:"#ff9900",
                     width:3
                },
				 xAxis:{
		     title:"Year",
		     template:"#year#"
		},
                yAxis:{
		     title:"Sales per year"
	        },
 
	    })
 
           lineChart.parse(data,"json");
        }
	</script>
</head>
