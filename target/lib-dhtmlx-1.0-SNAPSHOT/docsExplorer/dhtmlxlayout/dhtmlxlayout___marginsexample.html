	var dhxLayout;
	function doOnLoad() {
		dhxLayout = new dhtmlXLayoutObject(document.body, "3L");
		dhxLayout.cont.obj._offsetTop = 10;
		dhxLayout.cont.obj._offsetHeight = -30;
		dhxLayout.setSizes();
	}
